\gdef\ComponenteAutores{{%
            \flushright%
            \ABNTEXchapterfont%
            {\ValorDoNomeCompletoDoAutor{}}
            \par
        }}

\gdef\ComponenteTituloESubtitulo{{%
            \centering%
            \ABNTEXchapterfont%
            \textbf{\ValorDoTitulo{}}%
            \ifDefinidoSubtitulo%
                {\textbf{:} \ValorDoSubtitulo{}}%
            \fi
            \par
        }}

\gdef\ComponenteTituloESubtituloEstrangeiros{{%
            \centering%
            \ABNTEXchapterfont%
            \textit{
                \textbf{\ValorDoTituloEstrangeiro{}}%
                \ifDefinidoSubtituloEstrangeiro%
                    {\textbf{:} \ValorDoSubtituloEstrangeiro{}}%
                \fi
            }%
            \par
        }}

\gdef\ComponenteLocalEData{{%
            \center%
            \ABNTEXchapterfont%
            \ValorDaCidade{}%
            \ifDefinidoEstado%
                {,~\ValorDaSiglaDoEstado{}}%
            \fi
            \par
            \ValorDoAnoDeSubmissao{}
            \\
        }}

\gdef\ComponenteFontePropria{
    \fonte{Elaborado pelos autores (\ValorDoAnoDeSubmissao{}).}
}
